CORS는 브라우저가 다른 도메인으로 직접 API 호출하는 것을 막는 보안 정책입니다.
Cloudflare Worker는 무료 서버로 동작해 브라우저 대신 거래소 API를 호출합니다.
이 Worker 하나로 CoinEx + 모든 존버 거래소 API를 모두 처리합니다.
무료 플랜: 하루 10만 요청 · 배포 약 5분 소요
① Worker 배포 및 URL 설정
cloudflare.com 무료 가입 → Workers & Pages → Create Application → Hello World
아래 코드 전체 복사 → Worker 에디터에 붙여넣기 → 배포(Deploy) → URL 입력 → 테스트
WORKER 코드 — 멀티 거래소 지원 (CoinEx + Binance + Bybit + BingX + Bitget + HTX + OKX + Pionex)
export default {
  async fetch(request) {
    // CORS preflight
    if (request.method === 'OPTIONS') {
      return new Response(null, { headers: {
        'Access-Control-Allow-Origin': '*',
        'Access-Control-Allow-Methods': 'GET, POST, DELETE',
        'Access-Control-Allow-Headers': '*',
      }});
    }

    const url = new URL(request.url);
    // ?target=coinex  → https://api.coinex.com
    // ?target=binance → https://api.binance.com
    // ?target=bybit   → https://api.bybit.com
    // ?target=bingx   → https://open-api.bingx.com
    // ?target=bitget  → https://api.bitget.com
    // ?target=htx     → https://api.huobi.pro
    // ?target=okx     → https://www.okx.com
    // ?target=pionex  → https://api.pionex.com
    const hosts = {
      coinex:  'https://api.coinex.com',
      binance: 'https://api.binance.com',
      bybit:   'https://api.bybit.com',
      bingx:   'https://open-api.bingx.com',
      bitget:  'https://api.bitget.com',
      htx:     'https://api.huobi.pro',
      okx:     'https://www.okx.com',
      pionex:  'https://api.pionex.com',
    };

    const target = url.searchParams.get('target') || 'coinex';
    const baseHost = hosts[target] || hosts.coinex;
    url.searchParams.delete('target');

    const apiUrl = baseHost + url.pathname + (url.search ? url.search : '');

    // Forward all original headers
    const headers = {};
    for (const [k, v] of request.headers.entries()) {
      if (!['host','content-length'].includes(k.toLowerCase())) {
        headers[k] = v;
      }
    }

    const res = await fetch(apiUrl, {
      method: request.method,
      headers,
      body: request.method !== 'GET' ? await request.text() : undefined,
    });

    const data = await res.text();
    return new Response(data, {
      status: res.status,
      headers: {
        'Content-Type': res.headers.get('Content-Type') || 'application/json',
        'Access-Control-Allow-Origin': '*',
      }
    });
  }
};
